NAME
	titles - search and list matching titles to keyword

	authors - search and list matching authors to keyword

	years - search and list books which were published within a range

SYNOPSIS
	titles [OPTION] [SEARCH TERM]

	authors [OPTION] [SEARCH TERM]

	years [OPTION] [YEAR RANGE]

DESCRIPTION
	-a, -ascending [DEFAULT]
		lists the displayed results in ascending order 
	-d, -descending
		lists the displayed results in descending order
	-b, -book [DEFAULT]
		sorts by book title
	-n, -name
		sorts by author's name
	-y, -year
		sorts by publication year
